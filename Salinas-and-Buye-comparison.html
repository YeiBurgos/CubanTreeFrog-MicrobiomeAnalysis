# ---------- Function to clean dataset ----------
make_physeq <- function(otu_df, tax_df, meta_df, location){
  rownames(otu_df) <- otu_df$Group
  otu_mat <- t(otu_df[, -(1:3)])
  colnames(otu_mat) <- toupper(trimws(colnames(otu_mat)))
  
  # taxonomy
  library(tidyr)
  tax_sep <- tax_df %>%
    separate(Taxonomy, into=c("Kingdom","Phylum","Class","Order","Family","Genus"), 
             sep=";", fill="right", extra="warn")
  tax_clean <- tax_sep %>%
    select(any_of(c("OTU","Kingdom","Phylum","Class","Order","Family","Genus"))) %>%
    mutate(across(where(is.character), ~gsub("\\(\\d+\\)|k__|p__|c__|o__|f__|g__", "", .)))
  rownames(tax_clean) <- tax_clean$OTU
  tax_mat <- as.matrix(tax_clean[,-1])
  
  # metadata
  rownames(meta_df) <- toupper(trimws(meta_df$Group))
  colnames(meta_df)[colnames(meta_df) == "Sample"] <- "SampleType"
  meta_df$SampleType <- as.factor(meta_df$SampleType)
  meta_df$Location <- location
  
  # match samples
  shared <- intersect(colnames(otu_mat), rownames(meta_df))
  otu_mat <- otu_mat[, shared, drop=FALSE]
  meta_df <- meta_df[shared, , drop=FALSE]
  
  phyloseq(otu_table(otu_mat, taxa_are_rows=TRUE),
           tax_table(tax_mat),
           sample_data(meta_df))
}
